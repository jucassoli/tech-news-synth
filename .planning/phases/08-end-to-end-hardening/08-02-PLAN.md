```python
def term_jaccard(a: dict, b: dict) -> float:
    set_a, set_b = set(a.keys()), set(b.keys())
    if not set_a or not set_b:
        return 0.0
    return len(set_a & set_b) / len(set_a | set_b)
```

    1. **Create `scripts/cutover_verify.py`** with structure:
       ```python
       #!/usr/bin/env python3
       from __future__ import annotations
       import argparse, sys
       from datetime import UTC, datetime, timedelta
       from pathlib import Path
       from sqlalchemy import func, select
       from tech_news_synth.config import load_settings
       from tech_news_synth.db.session import init_engine, SessionLocal
       from tech_news_synth.db.models import Cluster, Post

       BASELINE_COST_24H_USD = 0.3612  # 12 × $0.03 + 12 × $0.0001 (from x-api-baseline.md)


       def term_jaccard(a: dict, b: dict) -> float:
           set_a, set_b = set(a.keys()), set(b.keys())
           if not set_a or not set_b:
               return 0.0
           return len(set_a & set_b) / len(set_a | set_b)


       def compute_verdict(since: datetime, jaccard_threshold: float, cost_multiplier: float, session) -> dict:
           now = datetime.now(UTC)
           since_plus_24h = since + timedelta(hours=24)
           window_start_48h = since - timedelta(hours=48)

           posted_24h = session.execute(
               select(func.count(Post.id))
               .where(Post.status == "posted", Post.posted_at >= since, Post.posted_at < since_plus_24h)
           ).scalar_one()

           rows = session.execute(
               select(Post.id, Post.posted_at, Post.cluster_id, Cluster.centroid_terms)
               .join(Cluster, Post.cluster_id == Cluster.id, isouter=True)
               .where(Post.status == "posted", Post.posted_at >= window_start_48h)
               .order_by(Post.posted_at)
           ).all()
           suspects = []
           for i, (id_a, ts_a, cid_a, terms_a) in enumerate(rows):
               for id_b, ts_b, cid_b, terms_b in rows[i+1:]:
                   if terms_a and terms_b:
                       sim = term_jaccard(terms_a, terms_b)
                       if sim >= jaccard_threshold:
                           suspects.append({
                               "post_a": id_a, "post_b": id_b,
                               "posted_a": ts_a.isoformat(), "posted_b": ts_b.isoformat(),
                               "jaccard": round(sim, 3),
                           })

           cost_sum = session.execute(
               select(func.coalesce(func.sum(Post.cost_usd), 0.0))
               .where(Post.status == "posted", Post.posted_at >= since, Post.posted_at < since_plus_24h)
           ).scalar_one()

           cost_ok = cost_sum <= BASELINE_COST_24H_USD * cost_multiplier
           count_ok = posted_24h >= 12
           dups_ok = len(suspects) == 0
           verdict = "GO" if count_ok and dups_ok and cost_ok else "NO-GO"

           return {
               "since": since.isoformat(),
               "now": now.isoformat(),
               "posted_24h": posted_24h, "count_ok": count_ok,
               "jaccard_suspects": suspects, "dups_ok": dups_ok,
               "cost_sum_usd": float(cost_sum), "cost_baseline_usd": BASELINE_COST_24H_USD,
               "cost_multiplier": cost_multiplier, "cost_ok": cost_ok,
               "verdict": verdict,
           }


       def render_report(result: dict) -> str:
           # Markdown report — see template section
           ...


       def main() -> int:
           parser = argparse.ArgumentParser(prog="cutover_verify")
           parser.add_argument("--since", required=True, help="ISO datetime (e.g., 2026-04-15T00:00:00Z)")
           parser.add_argument("--report-path", default=".planning/intel/cutover-report.md")
           parser.add_argument("--jaccard-threshold", type=float, default=0.5)
           parser.add_argument("--cost-multiplier", type=float, default=2.0)
           args = parser.parse_args()

           since = datetime.fromisoformat(args.since.replace("Z", "+00:00")).astimezone(UTC)
           settings = load_settings()
           init_engine(settings)
           with SessionLocal() as s:
               result = compute_verdict(since, args.jaccard_threshold, args.cost_multiplier, s)

           report_md = render_report(result)
           Path(args.report_path).parent.mkdir(parents=True, exist_ok=True)
           with Path(args.report_path).open("a", encoding="utf-8") as f:
               f.write(report_md)

           print(report_md)
           return 0 if result["verdict"] == "GO" else 1


       if __name__ == "__main__":
           raise SystemExit(main())
       ```
