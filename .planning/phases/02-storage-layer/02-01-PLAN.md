2. Implement `src/tech_news_synth/db/hashing.py` (pure-functional, no DB, no logging):
   ```python
   from __future__ import annotations
   import hashlib
   from urllib.parse import urlsplit, urlunsplit, parse_qsl, urlencode

   _TRACKING_PARAM_PREFIXES = ("utm_",)
   _TRACKING_PARAMS_EXACT = frozenset({"gclid", "fbclid"})


   def canonicalize_url(url: str) -> str:
       """Return a deterministic canonical form: lowercase scheme+host,
       drop fragment, strip utm_*/gclid/fbclid, sort remaining query params."""
       parts = urlsplit(url.strip())
       scheme = parts.scheme.lower()
       netloc = parts.netloc.lower()
       # keep path as-is (case preserved)
       query_pairs = [
           (k, v)
           for k, v in parse_qsl(parts.query, keep_blank_values=True)
           if not k.startswith(_TRACKING_PARAM_PREFIXES)
           and k not in _TRACKING_PARAMS_EXACT
       ]
       query_pairs.sort(key=lambda kv: (kv[0], kv[1]))
       query = urlencode(query_pairs)
       # Drop fragment by passing "" as the fifth element.
       return urlunsplit((scheme, netloc, parts.path, query, ""))


   def article_hash(url: str) -> str:
       """Return 64-char hex SHA256 of canonicalize_url(url)."""
       canon = canonicalize_url(url)
       return hashlib.sha256(canon.encode("utf-8")).hexdigest()
   ```

   ```python
   from __future__ import annotations
   from collections.abc import Iterator
   from contextlib import contextmanager

   from sqlalchemy import Engine, create_engine
   from sqlalchemy.orm import Session, sessionmaker

   from tech_news_synth.config import Settings
   from tech_news_synth.logging import get_logger

   log = get_logger(__name__)

   _engine: Engine | None = None
   _SessionLocal: sessionmaker[Session] | None = None


   def init_engine(settings: Settings) -> Engine:
       """Initialize module-level engine + SessionLocal. Idempotent (safe to call twice)."""
       global _engine, _SessionLocal
       if _engine is not None:
           return _engine
       _engine = create_engine(
           settings.database_url,
           pool_pre_ping=True,
           pool_size=5,
           future=True,
       )
       _SessionLocal = sessionmaker(bind=_engine, expire_on_commit=False, class_=Session)
       log.info("db_engine_initialized", pool_size=5)
       return _engine


   def SessionLocal() -> Session:
       """Return a new Session. init_engine(settings) must have been called."""
       if _SessionLocal is None:
           raise RuntimeError("init_engine(settings) must be called before SessionLocal()")
       return _SessionLocal()


   @contextmanager
   def get_session() -> Iterator[Session]:
       """Yield a Session; caller is responsible for commit()/rollback()."""
       session = SessionLocal()
       try:
           yield session
       finally:
           session.close()


   def _reset_engine_for_tests() -> None:
       """Test-only helper; never called in production code."""
       global _engine, _SessionLocal
       _engine = None
       _SessionLocal = None
   ```

```python
import numpy as np

def test_float32_bytes_roundtrip():
    original = np.asarray([0.1, 0.2, 0.3, -0.5, 1e-6], dtype=np.float32)
    blob = original.tobytes()
    restored = np.frombuffer(blob, dtype=np.float32)
```
