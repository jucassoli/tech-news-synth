   ```python
   from logging.config import fileConfig

   from alembic import context
   from sqlalchemy import engine_from_config, pool

   from tech_news_synth.config import load_settings
   from tech_news_synth.db.base import Base
   from tech_news_synth.db import models  # noqa: F401 — register models on Base.metadata

   config = context.config
   if config.config_file_name is not None:
       fileConfig(config.config_file_name)

   _settings = load_settings()
   config.set_main_option("sqlalchemy.url", _settings.database_url)

   target_metadata = Base.metadata


   def run_migrations_offline() -> None:
       url = config.get_main_option("sqlalchemy.url")
       context.configure(
           url=url,
           target_metadata=target_metadata,
           literal_binds=True,
           dialect_opts={"paramstyle": "named"},
           compare_type=True,
       )
       with context.begin_transaction():
           context.run_migrations()


   def run_migrations_online() -> None:
       connectable = engine_from_config(
           config.get_section(config.config_ini_section, {}),
           prefix="sqlalchemy.",
           poolclass=pool.NullPool,
       )
       with connectable.connect() as connection:
           context.configure(
               connection=connection,
               target_metadata=target_metadata,
               compare_type=True,
               compare_server_default=True,
           )
           with context.begin_transaction():
               context.run_migrations()


   if context.is_offline_mode():
       run_migrations_offline()
   else:
       run_migrations_online()
   ```

```python
def _drop_all_tables(dsn: str) -> None:
    eng = create_engine(dsn, future=True)
    with eng.begin() as conn:
        conn.execute(text("DROP SCHEMA public CASCADE; CREATE SCHEMA public;"))
    eng.dispose()


def test_upgrade_downgrade_upgrade_roundtrip(alembic_cfg):
    """STORE-01: downgrade path exists for the latest migration."""
    dsn = _test_database_url()
    _drop_all_tables(dsn)

    # Upgrade to head
    command.upgrade(alembic_cfg, "head")
    eng = create_engine(dsn, future=True)
    insp = inspect(eng)
    assert TABLES.issubset(set(insp.get_table_names()))
    assert "alembic_version" in insp.get_table_names()

    command.downgrade(alembic_cfg, "-1")
    insp = inspect(eng)
    remaining = set(insp.get_table_names())
    assert TABLES.isdisjoint(remaining), f"tables survived downgrade: {remaining & TABLES}"

    # Re-upgrade
    command.upgrade(alembic_cfg, "head")
    insp = inspect(eng)
    assert TABLES.issubset(set(insp.get_table_names()))
    eng.dispose()
```

```python
from __future__ import annotations
from collections.abc import Sequence
from typing import TypedDict
from datetime import datetime

from sqlalchemy import select
from sqlalchemy.dialects.postgresql import insert as pg_insert
from sqlalchemy.orm import Session

from tech_news_synth.db.models import Article


class ArticleRow(TypedDict, total=False):
    source: str
    url: str
    canonical_url: str
    title: str
    summary: str | None
    published_at: datetime | None
    article_hash: str
    etag: str | None
    last_modified: str | None


def upsert_batch(session: Session, rows: Sequence[ArticleRow]) -> int:
    if not rows:
        return 0
    stmt = (
        pg_insert(Article)
        .values(list(rows))
        .on_conflict_do_nothing(index_elements=["article_hash"])
    )
    result = session.execute(stmt)
    return result.rowcount or 0


def get_by_hash(session: Session, article_hash: str) -> Article | None:
    return session.execute(
        select(Article).where(Article.article_hash == article_hash)
    ).scalar_one_or_none()
```

```python
from datetime import UTC, datetime
from typing import Any

from sqlalchemy.orm import Session

def start_cycle(session: Session, cycle_id: str) -> RunLog:

    session.flush()

def finish_cycle(
    session: Session,
    cycle_id: str,
    status: str,
    counts: dict[str, Any] | None = None,
    notes: str | None = None,
) -> RunLog:
    row = session.execute(

    ).scalar_one()  # raises if not found

    if notes is not None:

    session.flush()
    return row
```
